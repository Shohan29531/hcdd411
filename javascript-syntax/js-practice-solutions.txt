Problem 8 — Grade Classifier
---------------------------------------------------------------
function getGrade(score) {
    if (score >= 90) {
        return "A";
    } else if (score >= 80) {
        return "B";
    } else if (score >= 70) {
        return "C";
    } else if (score >= 60) {
        return "D";
    } else {
        return "F";
    }
}

console.log(getGrade(95));   // A
console.log(getGrade(82));   // B
console.log(getGrade(74));   // C
console.log(getGrade(55));   // F

Problem 18 — Build a Class (BankAccount)
---------------------------------------------------------------
class BankAccount {
    constructor(balance) {
        this.balance = balance;
    }

    deposit(amount) {
        this.balance += amount;
    }

    withdraw(amount) {
        if (amount > this.balance) {
            console.log("Insufficient funds");
        } else {
            this.balance -= amount;
        }
    }

    getBalance() {
        return this.balance;
    }
}

let account = new BankAccount(1000);
account.deposit(500);
console.log(account.getBalance());  // 1500
account.withdraw(200);
console.log(account.getBalance());  // 1300
account.withdraw(2000);             // Insufficient funds

Problem 19 — Word Frequency Counter
---------------------------------------------------------------
function countWords(words) {
    let counts = {};

    words.forEach(function(word) {
        if (counts[word]) {
            counts[word]++;
        } else {
            counts[word] = 1;
        }
    });

    return counts;
}

let words = ["apple", "banana", "apple", "cherry", "banana", "apple"];
console.log(countWords(words));
// { apple: 3, banana: 2, cherry: 1 }

Problem 20 — Student Roster Manager
---------------------------------------------------------------
class Roster {
    constructor() {
        this.students = [];
    }

    addStudent(name, gpa) {
        this.students.push({ name: name, gpa: gpa });
    }

    removeStudent(name) {
        this.students = this.students.filter(function(student) {
            return student.name !== name;
        });
    }

    averageGPA() {
        if (this.students.length === 0) return 0;

        let total = 0;
        this.students.forEach(function(student) {
            total += student.gpa;
        });

        return total / this.students.length;
    }

    printRoster() {
        this.students.forEach(function(student) {
            console.log(student.name + " — GPA: " + student.gpa);
        });
    }
}

let roster = new Roster();
roster.addStudent("Alice", 3.8);
roster.addStudent("Bob", 3.2);
roster.addStudent("Carol", 3.9);
roster.printRoster();
console.log("Average GPA:", roster.averageGPA());
roster.removeStudent("Bob");
console.log("After removing Bob:");
roster.printRoster();
console.log("New Average GPA:", roster.averageGPA());
