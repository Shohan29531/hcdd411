Why Express is Needed

Node's built-in http module works, but even a small app fills up with repeated plumbing — matching methods, matching URLs, setting headers, structuring responses. Express removes all of that ceremony.

Raw Node http

Raw Node server
📄 server.js
  1. const http = require("http");
  3. const server = http.createServer((req, res) => {
  4. if (req.method === "GET" && req.url === "/") {
  5. res.writeHead(200, { "Content-Type": "text/plain" });
  6. res.end("Home page");
  7. } else if (req.method === "GET" && req.url === "/about") {
  8. res.writeHead(200, { "Content-Type": "text/plain" });
  9. res.end("About page");
  10. } else {
  11. res.writeHead(404, { "Content-Type": "text/plain" });
  12. res.end("Not found");
  13. }
  14. });
  16. server.listen(3000);

Express

Same job, clearer structure
📄 server.js
  1. const express = require("express");
  2. const app = express();
  4. app.get("/", (req, res) => {
  5. res.send("Home page");
  6. });
  8. app.get("/about", (req, res) => {
  9. res.send("About page");
  10. });
  12. app.listen(3000);
Mental model:

Express sits on top of Node and removes repeated ceremony — method matching, URL parsing, header boilerplate — so the route logic is the main thing you read.

Routes: Method + Path + Callback

All common HTTP methods
📄 server.js
  1. const express = require("express");
  2. const app = express();
  3. app.use(express.json());
  4. const PORT = 3000;
  6. // GET -- read a resource
  7. app.get("/products", (req, res) => {
  8. res.json([{ id: 1, name: "Widget" }]);
  9. });
  11. // POST -- create a new resource
  12. app.post("/products", (req, res) => {
  13. res.status(201).json({ created: true, body: req.body });
  14. });
  16. // PUT -- full replace of a resource
  17. app.put("/products/:id", (req, res) => {
  18. res.json({ replaced: req.params.id, body: req.body });
  19. });
  21. // PATCH -- partial update
  22. app.patch("/products/:id", (req, res) => {
  23. res.json({ updated: req.params.id, body: req.body });
  24. });
  26. // DELETE -- remove a resource
  27. app.delete("/products/:id", (req, res) => {
  28. res.json({ deleted: req.params.id });
  29. });
  31. app.listen(PORT, () => {
  32. console.log(`Listening on http://localhost:${PORT}`);
  33. });
PUT vs PATCH:

PUT replaces the entire resource — any fields you omit are gone. PATCH merges only the fields you send, leaving the rest unchanged.

Request Data: Params, Query, Body

req.params — named URL segments
📄 server.js
  1. const express = require("express");
  2. const app = express();
  3. const PORT = 3000;
  5. // Single param: GET /users/42
  6. app.get("/users/:id", (req, res) => {
  7. res.json(req.params); // { id: "42" }
  8. });
  10. // Multiple params: GET /users/5/posts/12
  11. app.get("/users/:userId/posts/:postId", (req, res) => {
  12. res.json(req.params); // { userId: "5", postId: "12" }
  13. });
  15. app.listen(PORT, () => {
  16. console.log(`Listening on http://localhost:${PORT}`);
  17. });

All param values arrive as strings — use Number() if you need a number.

req.query — query string
📄 server.js
  1. const express = require("express");
  2. const app = express();
  3. const PORT = 3000;
  5. app.get("/products", (req, res) => {
  6. // GET /products → req.query: {}
  7. // GET /products?category=books → req.query: { category: "books" }
  8. // GET /products?category=books&sort=price&order=asc
  9. // → req.query: { category: "books", sort: "price", order: "asc" }
  10. res.json(req.query);
  11. });
  13. app.listen(PORT, () => {
  14. console.log(`Listening on http://localhost:${PORT}`);
  15. });

Query keys are optional — always provide a fallback: req.query.sort || "name"

req.body — JSON payload
📄 server.js
  1. const express = require("express");
  2. const app = express();
  3. const PORT = 3000;
  5. app.use(express.json()); // must come before any route that reads req.body
  7. app.post("/orders", (req, res) => {
  8. // body: { item: "Pen", qty: 3, urgent: true }
  9. // req.body → { item: "Pen", qty: 3, urgent: true }
  10. res.json(req.body);
  11. });
  13. app.listen(PORT, () => {
  14. console.log(`Listening on http://localhost:${PORT}`);
  15. });

Forget express.json() and req.body is undefined — every field read from it will silently be undefined too.

Middleware: the Big Picture

Middleware sits between the incoming request and the final route handler. Each function can read, modify, or terminate the request — then pass control forward with next().

Request arrives
→
Middleware 1
→
Middleware 2
→
Route handler
→
Response

Middleware Anatomy and Order

A custom middleware function
📄 server.js
  1. const express = require("express");
  2. const app = express();
  3. const PORT = 3000;
  5. function logger(req, res, next) {
  6. console.log(`${req.method} ${req.url}`);
  7. next(); // hand off to the next step
  8. }
  10. app.use(logger); // runs for every request
  12. app.get("/demo", (req, res) => {
  13. res.send("Done");
  14. });
  16. app.listen(PORT, () => {
  17. console.log(`Listening on http://localhost:${PORT}`);
  18. });
Three parameters every middleware receives:
  • req — the incoming request object
  • res — the outgoing response object
  • next — a function that passes control to the next step
Important:

If middleware does not call next() and does not send a response, the request hangs forever. Every middleware must do one or the other.

2. Route-Specific Middleware

Attach middleware to selected routes only
📄 server.js
  1. const express = require("express");
  2. const app = express();
  3. const PORT = 3000;
  5. function requireAdmin(req, res, next) {
  6. if (req.query.role !== "admin") {
  7. return res.status(403).json({ error: "Admin only" });
  8. }
  9. next();
  10. }
  12. // requireAdmin runs only for this route
  13. app.get("/reports", requireAdmin, (req, res) => {
  14. res.send("Sensitive report data");
  15. });
  17. // This route is unaffected
  18. app.get("/public", (req, res) => {
  19. res.send("Anyone can see this");
  20. });
  22. app.listen(PORT, () => {
  23. console.log(`Listening on http://localhost:${PORT}`);
  24. });
Why this is useful:

Not every route needs the same gatekeeping. Passing middleware as a second argument keeps the restriction close to the route that needs it instead of running it globally.

4. Error-Handling Middleware

Must be placed after all routes
📄 server.js
  1. const express = require("express");
  2. const app = express();
  3. const PORT = 3000;
  5. app.get("/boom", (req, res, next) => {
  6. try {
  7. throw new Error("Something went wrong");
  8. } catch (err) {
  9. next(err); // forward error to error handler
  10. }
  11. });
  13. // Four parameters: Express knows this is an error handler
  14. app.use((err, req, res, next) => {
  15. console.error(err.message);
  16. res.status(500).json({ error: "Internal server error" });
  17. });
  19. app.listen(PORT, () => {
  20. console.log(`Listening on http://localhost:${PORT}`);
  21. });
Two details that matter:
  • The signature must have exactly four parameters: (err, req, res, next) — Express identifies error handlers by this
  • This handler belongs after all normal routes so errors bubble up to it

CORS: Why It Exists

The setup

A user is logged into bank.com in the browser.

The attack

A page on evil.com tries to read https://bank.com/api/balance.

Why this matters

The browser may send the user’s bank cookies, so without protection, evil.com could read private data.

What the malicious page tries to do
📄 evil-site.js
  1. fetch("https://bank.com/api/balance", {
  2. credentials: "include"
  3. })
  4. .then(res => res.json())
  5. .then(data => console.log(data));
What CORS does:

It lets bank.com tell the browser: only pages from bank.com may read this response.

Express example — allow only one browser origin
📄 server.js
  1. const express = require("express");
  2. const cors = require("cors");
  4. const app = express();
  5. const PORT = 3000;
  7. app.use(cors({ origin: "http://localhost:5173" }));
  9. app.get("/api/items", (req, res) => {
  10. res.json([{ id: 1, name: "Mouse" }]);
  11. });
  13. app.listen(PORT, () => {
  14. console.log(`Listening on http://localhost:${PORT}`);
  15. });

If request comes from 5173

The browser sees the allowed origin matches, so the page can read the response.

If request comes from 5174

The server may still respond, but the browser blocks that page’s JavaScript from reading the response.

Core idea

CORS does not stop the request from reaching the server. It controls which browser pages may read the response.

Putting It Together: Full CRUD API

server.js — inventory API with all HTTP methods
📄 server.js
  1. const express = require("express");
  2. const cors = require("cors");
  4. const app = express();
  5. app.use(cors());
  6. app.use(express.json());
  8. let items = [
  9. { id: 1, name: "Notebook", quantity: 4 },
  10. { id: 2, name: "Keyboard", quantity: 2 },
  11. ];
  13. // GET all items
  14. app.get("/items", (req, res) => {
  15. res.json(items);
  16. });
  18. // GET one item by id
  19. app.get("/items/:id", (req, res) => {
  20. const item = items.find(i => i.id === Number(req.params.id));
  21. if (!item) return res.status(404).json({ error: "Not found" });
  22. res.json(item);
  23. });
  25. // POST -- create a new item
  26. app.post("/items", (req, res) => {
  27. const newItem = { id: Date.now(), ...req.body };
  28. items.push(newItem);
  29. res.status(201).json(newItem);
  30. });
  32. // PUT -- full replace
  33. app.put("/items/:id", (req, res) => {
  34. const idx = items.findIndex(i => i.id === Number(req.params.id));
  35. if (idx === -1) return res.status(404).json({ error: "Not found" });
  36. items[idx] = { id: Number(req.params.id), ...req.body };
  37. res.json(items[idx]);
  38. });
  40. // PATCH -- partial update
  41. app.patch("/items/:id", (req, res) => {
  42. const item = items.find(i => i.id === Number(req.params.id));
  43. if (!item) return res.status(404).json({ error: "Not found" });
  44. Object.assign(item, req.body);
  45. res.json(item);
  46. });
  48. // DELETE -- remove an item
  49. app.delete("/items/:id", (req, res) => {
  50. const idx = items.findIndex(i => i.id === Number(req.params.id));
  51. if (idx === -1) return res.status(404).json({ error: "Not found" });
  52. const removed = items.splice(idx, 1)[0];
  53. res.json(removed);
  54. });
  56. // Error handler -- must be last
  57. app.use((err, req, res, next) => {
  58. console.error(err.message);
  59. res.status(500).json({ error: "Internal server error" });
  60. });
  62. app.listen(3000, () => console.log("Running on port 3000"));
MethodPathWhat it doesStatus
GET/itemsReturn all items200
GET/items/:idReturn one item or 404200 / 404
POST/itemsCreate a new item201
PUT/items/:idFull replace200 / 404
PATCH/items/:idPartial update200 / 404
DELETE/items/:idRemove and return item200 / 404
