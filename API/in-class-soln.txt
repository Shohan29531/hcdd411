// Practice 4: Fetch users then filter by name keyword

const keyword = "Leanne"; // Try changing this!

$.ajax({
    url: "https://jsonplaceholder.typicode.com/users",
    method: "GET",
    success: function (users) {
        // TODO: Make case-insensitive keyword and user.name comparisons
        const k = keyword.toLowerCase();

        // TODO: Filter users where name contains keyword
        const matched = users.filter(function (u) {
            return u.name.toLowerCase().includes(k);
        });

        // TODO: Log matched names (one per line)
        if (matched.length === 0) {
            console.log("No matches found for:", keyword);
        } else {
            matched.forEach(function (u) {
                console.log(u.name);
            });
        }

    },
    error: function (error) {
        console.error("Error:", error);
    }
});
